In the above cards we are using ng-content attributes. Please see below code.
Points to be remember.
  • 1. ng-content element without select attribute will pickup whole html available inside the product-card component.
  • 2. ng-content element with select attribute we can take only specific element like class or id. If it a class based pickup then we need to pass (.) or else (#).
  • If you will see example we are using with select with class based pickup.
import {{ '{' }} Component, OnInit } from '@angular/core';

@Component({{ '{' }}
    selector: 'product-card',
    template: `
    <div class="card">
        <div class="card-body">
            <ng-content select=".card-title"></ng-content>
            <a href="#" class="btn btn-primary">Go somewhere</a>
            <ng-content select=".card-text"></ng-content>
        </div>
    </div>
  `
})
export class ProductCardComponent implements OnInit {{ '{' }}
    flowType: string = "NI";

    constructor() {{ '{' }} }

    ngOnInit(): void {{ '{' }}
    }

}
